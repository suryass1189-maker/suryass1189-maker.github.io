/**
* Database Layer (IndexedDB Wrapper)
*/
const DB_NAME = 'StudyNotesDB';
const DB_VERSION = 1;
const db = {
instance: null,
init() {
return new Promise((resolve, reject) => {
const request = indexedDB.open(DB_NAME,
DB_VERSION);
request.onupgradeneeded = (e) => {
const db = e.target.result;
// Subjects: { id, name, date }
if (!db.objectStoreNames.contains('subjects'))
{
db.createObjectStore('subjects', {
keyPath: 'id' });
}
// Chapters: { id, subjectId, name, date }
if (!db.objectStoreNames.contains('chapters'))
{
const store =
db.createObjectStore('chapters', { keyPath: 'id' });
store.createIndex('subjectId',
'subjectId', { unique: false });
}
// Folders: { id, chapterId, name, date }
if (!db.objectStoreNames.contains('folders'))
{
const store =
db.createObjectStore('folders', { keyPath: 'id' });
store.createIndex('chapterId',
'chapterId', { unique: false });
}
// Files: { id, folderId, name, blob, size,
type, date }
if (!db.objectStoreNames.contains('files')) {
const store =
db.createObjectStore('files', { keyPath: 'id' });
store.createIndex('folderId', 'folderId',
{ unique: false });
}
};
request.onsuccess = (e) => {
this.instance = e.target.result;
resolve(this.instance);
};
request.onerror = (e) => reject(e);
});
},
async add(storeName, item) {
return new Promise((resolve, reject) => {
const tx = this.instance.transaction([storeName],
'readwrite');
const store = tx.objectStore(storeName);
const request = store.add(item);
request.onsuccess = () => resolve(request.result);
request.onerror = () => reject(request.error);
});
},
async getAll(storeName, indexName = null, indexValue =
null) {
return new Promise((resolve, reject) => {
const tx = this.instance.transaction([storeName],
'readonly');
const store = tx.objectStore(storeName);
let request;
if (indexName && indexValue !== null) {
const index = store.index(indexName);
request = index.getAll(indexValue);
} else {
request = store.getAll();
}
request.onsuccess = () => resolve(request.result);
request.onerror = () => reject(request.error);
});
},
async delete(storeName, id) {
return new Promise((resolve, reject) => {
const tx = this.instance.transaction([storeName],
'readwrite');
const store = tx.objectStore(storeName);
const request = store.delete(id);
request.onsuccess = () => resolve();
request.onerror = () => reject(request.error);
});
},
async update(storeName, item) {
return new Promise((resolve, reject) => {
const tx = this.instance.transaction([storeName],
'readwrite');
const store = tx.objectStore(storeName);
const request = store.put(item);
request.onsuccess = () => resolve();
request.onerror = () => reject(request.error);
});
},
async get(storeName, id) {
return new Promise((resolve, reject) => {
const tx = this.instance.transaction([storeName],
'readonly');
const store = tx.objectStore(storeName);
const request = store.get(id);
request.onsuccess = () => resolve(request.result);
request.onerror = () => reject(request.error);
});
}
};

openModal(modalId) {
let targetId = modalId;
let isVirtual = false;
// Handle Virtual IDs by resolving them to the generic
'modal-input'
if (['modal-subject', 'modal-chapter',
'modal-folder'].includes(modalId)) {
targetId = 'modal-input';
isVirtual = true;
}
const modal = document.getElementById(targetId);
if (!modal) {
console.error(`Modal element with ID '${targetId}'
not found.`);
return;
}
const content = modal.querySelector('.modal-content');
modal.classList.remove('hidden');
modal.classList.add('flex');
// Configure Modal based on Type
if (targetId === 'modal-input') {
// Reset form state first
document.getElementById('input-form').reset();
document.getElementById('input-id').value = '';
if (isVirtual) {
// Explicit configuration from the button
click
if (modalId === 'modal-subject') {
document.getElementById('input-type').value = 'subject';
document.getElementById('modal-title').textContent = 'New Subject';
document.getElementById('input-name').placeholder = 'e.g., Physics';
} else if (modalId === 'modal-chapter') {
document.getElementById('input-type').value = 'chapter';
document.getElementById('modal-title').textContent = 'New Chapter';
document.getElementById('input-name').placeholder = 'e.g.,
Kinematics';
} else if (modalId === 'modal-folder') {
document.getElementById('input-type').value = 'folder';
document.getElementById('modal-title').textContent = 'New Folder';
document.getElementById('input-name').placeholder = 'e.g., MCQ Bank';
}
} else {
// Context Inference (Fallback)
// Used when calling openModal('modal-input')
directly (e.g., from FAB)
if (this.state.view === 'home') {
document.getElementById('input-type').value = 'subject';
document.getElementById('modal-title').textContent = 'New Subject';
document.getElementById('input-name').placeholder = 'e.g., Physics';
} else if (this.state.view === 'subject') {
document.getElementById('input-type').value = 'chapter';
document.getElementById('modal-title').textContent = 'New Chapter';
document.getElementById('input-name').placeholder = 'e.g.,
Kinematics';
} else if (this.state.view === 'chapter') {
document.getElementById('input-type').value = 'folder';
document.getElementById('modal-title').textContent = 'New Folder';
document.getElementById('input-name').placeholder = 'e.g., MCQ Bank';
}

document.getElementById('input-name').focus(), 100);
}

content.classList.remove('scale-95',
'opacity-0');

'opacity-100');

// --- Utilities ---
formatSize(bytes) {
if (bytes === 0) return '0 B';
const k = 1024;
const sizes = ['B', 'KB', 'MB', 'GB'];
const i = Math.floor(Math.log(bytes) / Math.log(k));
return parseFloat((bytes / Math.pow(k, i)).toFixed(1))
+ ' ' + sizes[i];
},
async updateStorageUsage() {
if (navigator.storage && navigator.storage.estimate)
{
const estimate = await
navigator.storage.estimate();
const usedMB = (estimate.usage / (1024 *
1024)).toFixed(1);
document.getElementById('storage-used').textContent = `${usedMB} MB`;
}
